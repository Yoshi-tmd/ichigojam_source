// カミナリ避けゲーム
// どこかで拾ったゲーム

10 CLS:CLP:X=14:Y=23:W=200:S=0
20 SCROLL 2
30 C=SCR(X,Y)
40 LC X,Y:?CHR$(250);
50 IF RND(2)!=1 GOTO 70
60 LC RND(30),0:?CHR$(242);
70 X=X-(BTN(28) AND X>0)+(BTN(29) AND X<28)
80 S=S+1
90 CLK
100 WAIT W/10
110 W=W-1:IF W<10 W=9
120 IF C!=242 GOTO 20
130 LC X,Y:?"X";
140 BEEP 31,60
150 LC 11,13:?"GAME OVER"
160 ?S;" POINTS"
