// 糸通しゲーム
// 距離センサーで操作するゲーム
// 応用クラスのMISSLIMゲームと類似

1 'ITOTOSHI
10 '?ANA(2)-350:WAIT 10:GOTO 10
20 CLS:C=0:Y=11
23 IF C%10==0 GSB 230
25 A=ANA(2)-350
26 IF A>600 A=600:IF A<0 A=0
27 Y=A/30
28 SCROLL 3
29 LC 2,Y:?"ǳ";
30 IF SCR(3,Y)==ASC("|") END
35 C=C+1
38 IF C>200 END
40 WAIT 5:GOTO 23
230 I=0
240 LC 30,I:?"|"
250 I=I+1
260 IF I<22 GOTO 240
270 J=3+RND(13):D=J
280 LC 30,J:?" "
290 J=J+1
300 IF J<D+5 GOTO 280
310 RTN
