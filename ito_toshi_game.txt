// 糸通しゲーム Ver.2
// キーボード操作の糸通し

1 'ITOTOSHI VER.2
20 CLS:C=0:Y=11
23 IF C%10==0 GSB 230
25 N=1-BTN(30)*2
27 Y=Y+N+BTN(31)
28 SCROLL 3
29 LC 2,Y:?"ǳ";
30 IF SCR(3,Y)==ASC("|") END
35 C=C+1
38 IF C>200 END
39 IF Y>21 END
40 WAIT 5:GOTO 23
230 I=0
240 LC 30,I:?"|"
250 I=I+1
260 IF I<22 GOTO 240
270 J=3+RND(13):D=J
280 LC 30,J:?" "
290 J=J+1
300 IF J<D+5 GOTO 280
310 RTN
