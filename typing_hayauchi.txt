[タイピング早打ちゲーム]

10 S=0:M=0
20 C=65+RND(26):X=RND(32):Y=0
30 CLS:LC X,Y:?CHR$(C);
40 LC 0,23:?"SCORE=";S,"MISS=";M;
50 K=INKEY()
60 IF K IF K=C BEEP5:S=S+25-Y:GOTO20:ELSE:BEEP50:IF S<3000 S=S-S/10:ELSE GOTO100
70 WAIT 10-S/300
80 Y=Y+1:IF Y<23 GOTO30
90 M=M+1:IF M<3 THEN PLAY"A16G16A2R8G16F16E16D16D-8D4":WAIT150:GOTO20
100 PLAY"C4C8.C16C4E-8.D16D8.C16C8.>B16<C4"
110 FOR I=0TO24:SCROLL0:WAIT5:NEXT
120 LC 10,10:?"GAME OVER":LC 10,12:?"SCORE=";S
