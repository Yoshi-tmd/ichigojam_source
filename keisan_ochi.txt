// 計算落ちゲーム
// 落ちてくる数字を拾って、自分の数字と足して10になるようにする

1 'OCHI GAME
10 CLS:CLT
20 X=16:C=0
21 W=RND(31):Z=0:N=RND(10)
23 'CLS
30 IF C%4==0 GSB 300
38 Z=Z+1
40 X=X-(BTN(28))+(BTN(29))
50 LC X,22:?"Ǻ"
60 LC X,21:?N
70 M=SCR(X,20)
80 IF M M=M-48:N=N+M:GSB 400
90 IF TICK()>600 END
100 C=C+1
110 SCROLL 2
120 WAIT 6:GOTO 23
300 LC RND(31),0:?RND(10)
310 RTN
400 IF N<10 RTN
410 IF N==10 BEEP 10,10
420 IF N>10 BEEP 60,10
430 N=RND(10)
440 RTN
