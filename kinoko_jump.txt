20 X=15:Y=20
30 LC X-1,Y:?".";

42 X=X+V:Y=Y+U
45 IF (V!=0)*(V>0) V=V-1
46 IF (V!=0)*(V<0) V=V+1
47 U=U+1
50 K=INKEY()
60 IF K==28 V=-3
70 IF K==29 V=3
75 IF K==32 U=-3
80 IF X<0 X=0
82 IF X>31 X=31
84 IF Y>20 Y=20
90 LC X,Y:?CHR$(250);
100 IF !RND(5) GSB 400
200 SCROLL 3:WAIT 1:GOTO 30
400 L=20
410 LC 31,L:?CHR$(226)
420 L=L-1
430 IF (L==10)+(!RND(5)) RTN
