# ジャンプイチゴキャッチゲーム

10 A=0:B=12:S=0
20 C=RND(26)+3
30 CLS
40 ?CHR$(255);S
45 IF S==5 END
50 LC C,9:?CHR$(255)
60 LC A,B:?CHR$(236)
70 A=A+1
80 IF A>30 A=0
90 IF B==9 B=B+3
100 IF INKEY()==32 B=B-3
110 IF A==C AND B==9 S=S+1:GOTO20
120 WAIT5:GOTO30
