■数字消しゲーム
10 CLS:CLP:CLV
20 X=15:Y=11:B=0:S=0:T=1800
30 FOR I=1 TO 32*23:?RND(4)+1;:NEXT
35 CLT
40 C=SCR(X,Y)-48
100 LC X,Y
110 B=B+1:IF B%40<20 ?C; ELSE ?CHR$(1);
120 K=INKEY()
130 IF K==0 GOTO 100
140 E=X-(K==28)+(K==29)
150 F=Y-(K==30)+(K==31)
160 D=SCR(E,F)-48
170 IF D<0 OR (D+C)>10 GOTO 100
180 IF (D+C)==10 S=S+1:C=0:D=0:LC E,F:?" ";
185 IF TICK()>T GOTO 300
190 C=C+D
200 LC X,Y:?" ";
210 X=E:Y=F
220 GOTO 100
300 LC 2,23:?"TIME OVER SCORE:";S:END
